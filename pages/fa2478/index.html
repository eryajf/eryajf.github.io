# и§Ҷйў‘

жң¬ж–ҮеҶ…е®№йҖҡиҝҮBз«ҷиҝӣиЎҢдәҶзӣҙж’ӯеҲҶдә«пјҢеҸҜзӣҙжҺҘзңӢи§Ҷйў‘дәҶи§Јжң¬ж–ҮеҶ…е®№пјҡ

# еүҚиЁҖ

еҸӨд»ЈжІ»еӯҰжҖ»з»“жңүдәәз”ҹдёүеўғз•ҢпјҢеңЁжҲ‘зңӢжқҘпјҢJenkins дҪҝз”ЁжҺҘе…ҘдәҰжңүдёүдёӘйҳ¶ж®өпјҢиҝҷд№ҹжҳҜжҜҸдёҖдёӘиҝҗз»ҙдәәе‘ҳеә”з”Ё Jenkins жүҖеҝ…йЎ»иҰҒз»ҸеҺҶзҡ„гҖӮ

stage("зүҲжң¬еҸ·еҶҷе…Ҙ") {
    when {
        environment name: 'MODE',value: 'DEPLOY'
    }
    steps {
        script {
            try {
                env.FILE=sh (script:"ls ${VERSION_FILE}",returnStatus: true)
                if("${env.FILE}" != "0") {
                    sh "echo \"${VERSION_KEY}=${_VERSION}\" > ${VERSION_FILE}"
                }else {
                    sh 'sed -i "s#=#&${_VERSION},#" ${VERSION_FILE}'
                }
                env.NUMBER=sh (script: 'grep -o , ${VERSION_FILE} | wc -l', returnStdout: true).trim()
                // еҲӨж–ӯзүҲжң¬еҸ·жҳҜеҗҰдёә10дёӘ
                if("${NUMBER}" == "10") {
                    sh '''
                        sed -i "s#,`cut -d, -f11 ${VERSION_FILE}`##" ${VERSION_FILE}
                    '''
                }
            }catch(exc) {
            env.REASON = "зүҲжң¬еҸ·еҶҷе…ҘеҮәй”ҷ"
            throw(exc)
            }
        }
    }
}

# ж–°е»әйЎ№зӣ®

еҰӮдёҠз§Қз§Қй…ҚзҪ®е®үжҺ’еҲ°дҪҚд№ӢеҗҺпјҢж–°е»әйЎ№зӣ®е°ҶдјҡжҳҜдёҖдёӘеҲҶй’ҹзә§еҲ«зҡ„ж“ҚдҪңпјҢдёәдәҶеұ•зӨәиҝҷз§ҚдҪҝз”Ёж–№ејҸзҡ„дјҳеҠҝпјҢжҲ‘д№ҹжЁЎжӢҹе®һи·өж–°е»әдёҖдёӘйЎ№зӣ®пјҢзңӢзңӢеӨ§жҰӮзҡ„жөҒзЁӢгҖӮ

# иҺ·еҫ—зҙ жқҗ

зҙ жқҗдәҰеҚіеј•еҜјж–Үд»¶дёӯжүҖйңҖиҰҒеЎ«е……жӣҝжҚўзҡ„еҶ…е®№пјҢиҝҷйҮҢзҡ„еҶ…е®№е»әи®®еңЁдјҒдёҡдёӯеҒҡжҲҗе·ҘеҚ•жЁЎжқҝпјҢжҜ”еҰӮиҝҷдёӘеүҚз«ҜйЎ№зӣ®пјҢе°ұеә”иҜҘеҒҡдёҖдёӘеүҚз«Ҝзұ»зҡ„е·ҘеҚ•жЁЎжқҝпјҢдёҡеҠЎж–№йңҖиҰҒдёҠзәҝж–°зҡ„йЎ№зӣ®пјҢе°ұеҸҜд»ҘжҢүз…§е·ҘеҚ•жЁЎжқҝпјҢжҠҠжҲ‘д»¬е…іеҝғзҡ„еҶ…е®№жҸҗдәӨиҝҮжқҘпјҢз”ұиҝҗз»ҙиҪ¬еҢ–жҲҗ Jenkins йЈҺж јзҡ„еј•еҜјж–Үд»¶еҶ…е®№гҖӮ

еҪ“з„¶иҝҳеҸҜд»ҘеҶҚеҫҖе№іеҸ°еҢ–ж–№еҗ‘иҝӣдёҖжӯҘпјҢе°ұжҳҜзӣҙжҺҘе°Ҷж–Үд»¶зҡ„з”ҹдә§дёҺе·ҘеҚ•жЁЎжқҝжү“йҖҡпјҢеҪ“дёҡеҠЎж–№жҸҗдәӨдәҶе·ҘеҚ•д№ӢеҗҺпјҢзӣҙжҺҘз”ҹжҲҗеј•еҜјж–Үд»¶зҡ„еҶ…е®№гҖӮ

жӢҝдёҠиҫ№зҡ„йЎ№зӣ®дёҫдҫӢеӯҗпјҢзҺ°еңЁйңҖиҰҒдёҠзәҝиҜҘйЎ№зӣ®зҡ„йў„еҸ‘зҺҜеўғпјҢеҲҷеј•еҜјж–Үд»¶и·Ҝеҫ„дёә ops/pre-eryajf-blog.jenkins пјҢеҶ…е®№еҰӮдёӢпјҡ

@Library('global-shared-library@main') _

def map = [:]
    // е®ҡд№үйЎ№зӣ®жһ„е»әиҝҗиЎҢзҡ„ NODE ,ж №жҚ®е®һйҷ…жғ…еҶөиҝӣиЎҢи°ғж•ҙ
    map.put('RUN_NODE','master')
    // йңҖиҰҒдҝ®ж”№жӯӨеӨ„пјҢе®ҡд№үйЎ№зӣ®еҗҚз§°
    map.put('SERVICE_NAME','pre-eryajf-blog.eryajf.net')
    // е®ҡд№үwebrootзӣ®еҪ•,дёҖиҲ¬е»әи®®/data/www/${SERVICE_NAME}дёӢ
    map.put('WEBROOT_DIR','/data/www/${SERVICE_NAME}')
    // е®ҡд№үйЎ№зӣ®й»ҳи®Өзҡ„еҲҶж”ҜпјҢж №жҚ®е®һйҷ…жғ…еҶөи°ғж•ҙ
    map.put('DEFAULT_BRANCH','main')
    // е®ҡд№үйЎ№зӣ®gitең°еқҖ
    map.put('GIT_URL','https://jihulab.com/eryajf-jenkins/eryajf-blog.git')
    // е®ҡд№үдё»жңәйҖүйЎ№еҸӮж•°пјҢеӨҡеҸ°з”Ё\nеҲҶеүІ
    map.put('HOSTS','ALL\n172.19.192.132')
    // е®ҡд№үйЎ№зӣ®зј–иҜ‘е‘Ҫд»Ө
    map.put('BUILD_COMMAND','echo test')
    // е®ҡд№үйЎ№зӣ®йғЁзҪІд№ӢеҗҺжү§иЎҢзҡ„и„ҡжң¬,жіЁж„ҸжӯӨи„ҡжң¬жү§иЎҢдҪҚзҪ®дёә ${WEBROOT_DIR}
    map.put('FREE_COMMAND','chown -R www.www /data/www/${SERVICE_NAME}/')
    // е®ҡд№үеҝҪз•Ҙж–Үд»¶жҲ–зӣ®еҪ•пјҢеӨҡдёӘз”Ё \n еҲҶеүІ
    map.put('EXCLUDE_FILE','ansible_tmp\nansible_tmp@tmp\n.git\nnode_modules')
    // з”ЁдәҺжү“еҢ…зј–иҜ‘зҡ„еҹәзЎҖй•ңеғҸ
    map.put('BUILD_BASE_IMAGE','eryajf/node:10.6')
    // жҢҮе®ҡе°ҶиҰҒйғЁзҪІеҲ°иҝңзЁӢзҡ„зӣ®еҪ•,еҰӮжһңйғЁзҪІж №зӣ®еҪ•пјҢз”Ё . иЎЁзӨә
    map.put('PROJECT_FILE_PATH','.')
    // жҢҮе®ҡжңәеҷЁдәәkey
    map.put('ROBOT_KEY','6a781aaf-0cda-41ab-9bd2-ed81ee7fc7d2')
    // жҢҮе®ҡзүҲжң¬зҡ„и·Ҝеҫ„д»ҘеҸҠkeyпјҢдёҖиҲ¬дёҚйңҖиҰҒжӣҙж”№еҰӮдёӢдёӨйЎ№
    map.put('VERSION_KEY', "$JOB_BASE_NAME" )
    map.put('VERSION_FILE', "/jenkins_sync/version/$JOB_BASE_NAME" )
